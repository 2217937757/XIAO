import numpy as np
x = np.arange(0,10,0.1)
y = np.sin(x)
plt.plot(x, y, label='sin(x)')
plt.ylim(-1.5,1.5)
plt.xlabel('variable x')
plt.ylabel('value y')
plt.title('')
plt.grid()
plt.axhline(y=0.8,c='r')
plt.axvspan(xmin=4, xmax=6, facecolor='r', alpha=0.3) # x
plt.axhspan(ymin=-0.2, ymax=0.2, facecolor='y', alpha=0.3)  # y
plt.text(3.2, 0, 'sin(x)', color='r')
plt.annotate('maximum',xy=(np.pi/2, 1),xytext=(np.pi/2+1, 1),
             color='r',
             arrowprops=dict(arrowstyle='->',  color='r'))
plt.legend()
